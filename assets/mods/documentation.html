UI Activation

after.player.death = activate.ui("UI_lol").and.animate.from.right_top.to.center_top.elastic.out;
after.player.respawn = disable.ui("UI_lol");

// id or class both work:
// <div id="UI_lol">...</div>
// <div class="UI_lol">...</div>

Position Classes

ic  // inside center
il  // inside left
ir  // inside right
sl  // side left
sr  // side right

Animation Grammar

.and.animate.from.left.to.right.elastic.in

from / to supported points:
left, right, top, bottom, center,
left_top, top_left, right_top, top_right,
left_bottom, bottom_left, right_bottom, bottom_right

easing type:
ease, elastic, bounce, exponential, sine, back

easing direction:
in, out, inout

Timed Hooks

// fire 2 seconds after game.start event
after.game.start.for["2"].seconds = activate.mod.platformer;

// units: milliseconds, seconds, minutes, hours

Timed Actions + Then Chain

after.game.start = activate.mod.platformer.for["2"].minutes.then.activate("ui2");

// this enables platformer for 2 minutes,
// then runs named action "ui2"

Named Actions (def)

const ui2 = activate.ui("UI_lol").and.animate.from.right_top.to.center_top.elastic.out;
def("ui2", ui2);

if (attempt.count > larger.than["13"]) {
  after.player.death = activate.mod.rainbow;
} else {
  activate("ui2");
}

Attempt Helpers

attempt.count            // current attempt number
larger.than["13"]           // number helper (13)
smaller.than["5"]           // number helper (5)

`def ui2 = ...` is not valid JavaScript syntax. Use const ui2 = ... and then def("ui2", ui2).
